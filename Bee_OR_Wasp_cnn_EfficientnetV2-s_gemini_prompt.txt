import kagglehub
import shutil
import os

# 1. 데이터셋 다운로드 (임시 경로로 저장됨)
tmp_path = kagglehub.dataset_download("jerzydziewierz/bee-vs-wasp")
print("임시 저장 경로:", tmp_path)

# 2. 목적지 경로 설정 (현재 프로젝트 경로/dataset)
target_path = os.path.join(os.getcwd(), "dataset")

# 3. 폴더 이동 로직
try:
    if not os.path.exists(target_path):
        # 목적지 폴더가 없으면 바로 이동
        shutil.move(tmp_path, target_path)
        print(f"성공: 데이터를 {target_path}로 이동했습니다.")
    else:
        # 이미 폴더가 있는 경우 내부 파일들만 복사/이동
        print(f"알림: '{target_path}' 폴더가 이미 존재합니다. 내부 파일을 병합합니다.")
        for item in os.listdir(tmp_path):
            s = os.path.join(tmp_path, item)
            d = os.path.join(target_path, item)
            if os.path.isdir(s):
                shutil.copytree(s, d, dirs_exist_ok=True)
            else:
                shutil.copy2(s, d)
        print("성공: 파일 이동 완료.")
except Exception as e:
    print(f"오류 발생: {e}")
